public boolean tryLock(long waitTime, long leaseTime, TimeUnit unit) throws InterruptedException {
        long time = unit.toMillis(waitTime); // 剩余可等待时间（毫秒）
        long current = System.currentTimeMillis();
        long threadId = Thread.currentThread().getId();

        // 尝试加锁（执行 Lua 脚本）
        Long ttl = this.tryAcquire(waitTime, leaseTime, unit, threadId);

        // 抢锁成功，直接返回 true
        if (ttl == null) {
                return true;
        }

        // 抢锁失败，计算剩下的时间。如果已经没时间等了，直接返回 false
        time -= System.currentTimeMillis() - current;
        if (time <= 0L) {
                this.acquireFailed(waitTime, unit, threadId);
                return false;
            } else {
                current = System.currentTimeMillis();
                // 订阅该锁的频道
                CompletableFuture<RedissonLockEntry> subscribeFuture = this.subscribe(threadId);

                try {
                    // 阻塞等待订阅结果。如果在 time 时间内订阅都没成功，直接报错 
                    subscribeFuture.get(time, TimeUnit.MILLISECONDS);
                } catch (TimeoutException var21) {
                    // 订阅超时处理逻辑
                    if (!subscribeFuture.completeExceptionally(new RedisTimeoutException("Unable to acquire subscription lock after " + time + "ms. Try to increase 'subscriptionsPerConnection' and/or 'subscriptionConnectionPoolSize' parameters."))) {
                        subscribeFuture.whenComplete((res, ex) -> {
                            if (ex == null) {
                                this.unsubscribe(res, threadId);
                            }

                        });
                    }

                    this.acquireFailed(waitTime, unit, threadId);
                    return false;
                } catch (ExecutionException e) {
                    LOGGER.error(e.getMessage(), e);
                    this.acquireFailed(waitTime, unit, threadId);
                    return false;
                }

                boolean currentTime;
                try {
                    // 扣除订阅花掉的时间
                    time -= System.currentTimeMillis() - current;
                    if (time > 0L) {
                        do {
                            long currentTime = System.currentTimeMillis();
                            // 再次尝试抢锁
                            ttl = this.tryAcquire(waitTime, leaseTime, unit, threadId);
                            // 抢到了，结束
                            if (ttl == null) {
                                boolean var32 = true;
                                return var32;
                            }

                            // 没抢到，扣除抢锁花掉的时间
                            time -= System.currentTimeMillis() - currentTime;
                            if (time <= 0L) {
                                this.acquireFailed(waitTime, unit, threadId);
                                boolean var31 = false;
                                return var31;
                            }

                            // 拿到该锁对应的信号量（getLatch），如果在睡眠时间内锁被释放，进程会被getlatch叫醒
                            currentTime = System.currentTimeMillis();
                            if (ttl >= 0L && ttl < time) {
                                // 如果 ttl 比剩余等待时间短，就睡 ttl 毫秒
                                ((RedissonLockEntry)this.commandExecutor.getNow(subscribeFuture)).getLatch().tryAcquire(ttl, TimeUnit.MILLISECONDS);
                            } else {
                                // 否则睡到 waitTime 结束
                                ((RedissonLockEntry)this.commandExecutor.getNow(subscribeFuture)).getLatch().tryAcquire(time, TimeUnit.MILLISECONDS);
                            }
                            // 此时线程处于休眠状态，直到：
                            // 1. 睡够了（ttl 到期）
                            // 2. 有人解锁发了广播，Redisson 收到消息立刻 release 信号量，将线程唤醒
                            time -= System.currentTimeMillis() - currentTime;// 扣除睡觉花掉的时间
                        } while(time > 0L);//回循环继续抢锁

                        this.acquireFailed(waitTime, unit, threadId);
                        boolean var16 = false;
                        return var16;
                    }

                    this.acquireFailed(waitTime, unit, threadId);
                    currentTime = false;
                } finally {
                    // 无论成功失败，最后都要退订广播
                    this.unsubscribe((RedissonLockEntry)this.commandExecutor.getNow(subscribeFuture), threadId);
                }

                return currentTime;
            }
        }
    }

private RFuture<Long> tryAcquireAsync(long waitTime, long leaseTime, TimeUnit unit, long threadId) {
        RFuture<Long> ttlRemainingFuture;
        if (leaseTime > 0L) {
            ttlRemainingFuture = this.<Long>tryLockInnerAsync(waitTime, leaseTime, unit, threadId, RedisCommands.EVAL_LONG);
        } else {
            // 使用 Redisson 默认的 internalLockLeaseTime（默认 30秒）
            ttlRemainingFuture = this.<Long>tryLockInnerAsync(waitTime, this.internalLockLeaseTime, TimeUnit.MILLISECONDS, threadId, RedisCommands.EVAL_LONG);
        }

       // 处理一些本地缓存或同步状态，确保 Future 的状态是正确的
        CompletionStage<Long> s = this.handleNoSync(threadId, ttlRemainingFuture);
        ttlRemainingFuture = new CompletableFutureWrapper(s);
        // thenApply:当之前的任务完成后，拿着结果去执行下一步，并返回一个新的结果
        // ttlRemainingFuture 是 RFuture<Long>，当抢锁结果（ttlRemaining）回来时
        CompletionStage<Long> f = ttlRemainingFuture.thenApply((ttlRemaining) -> {
            if (ttlRemaining == null) {
                // 抢到了
                if (leaseTime > 0L) {
                    // 自己设置了过期时间
                    this.internalLockLeaseTime = unit.toMillis(leaseTime);
                } else {
                    //没设置过期时间，开启看门狗续期
                    this.scheduleExpirationRenewal(threadId);
                }
            }

            return ttlRemaining;
        });
        return new CompletableFutureWrapper(f);
    }

public void renewLock(String name, Long threadId, String lockName) {
        //reference：这是一个 AtomicReference（原子引用）
        //存放全局唯一的 LockTask，internalLockLeaseTime续期间隔，executor负责执行续期操作的线程池，batchSize一次合并多少个续期请求
        this.reference.compareAndSet((Object)null, new LockTask(this.internalLockLeaseTime, this.executor, this.batchSize));
        //LockTask 内部维护了一个 Map 或者 集合。它把你的锁信息加到这个“待续期清单”里，LockTask内部执行批量续期 Lua 脚本
        //在加入第一把锁后，RenewalTask（LockTask的父类）就把running设为true，开启schedule() 往时间轮里塞入一个 10s 后的闹钟，10s钟后执行run()方法，在这个方法里第一步调用renew()执行 Lua 脚本（LockTask中的buildChunk）给 Redis 里的锁续30s，第二步再次触发 schedule()，预约下一个 10秒。
        LockTask task = (LockTask)this.reference.get();
        task.add(name, lockName, threadId);
    }

public RFuture<Void> unlockAsync(long threadId) {
        // 为了防止同一个解锁请求被重复执行（幂等性），或者为了追踪这个异步任务的进度，Redisson 会给每个操作生成一个唯一的 ID
        String requestId = this.getServiceManager().generateId();

    }

private RFuture<Void> unlockAsync0(long threadId, String requestId) {
        // Lua 解锁脚本发给 Redis
        CompletionStage<Boolean> future = this.unlockInnerAsync(threadId, requestId);
        CompletionStage<Void> f = future.handle((res, e) -> {
            // 关闭看门狗，只要 Redis 给了回复（不管 res 是 0 还是 1），它都会去尝试清理本地的续期任务
            this.cancelExpirationRenewal(threadId, res);
            if (e != null) {
                if (e instanceof CompletionException) {
                    throw (CompletionException)e;
                } else {
                    throw new CompletionException(e);
                }
            } else if (res == null) {
                IllegalMonitorStateException cause = new IllegalMonitorStateException("attempt to unlock lock, not locked by current thread by node id: " + this.id + " thread-id: " + threadId);
                throw new CompletionException(cause);
            } else {
                return null;
            }
        });
        return new CompletableFutureWrapper(f);
    }
