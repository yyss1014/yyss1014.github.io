创建对象

构造器 说明
public File(String pathname) 根据文件路径创建对象
public File(String parent,String child) 根据父路径和子路径名字创建文件对象
public File(File parent,String child) 根据父路径对应文件对象和子路径名字创建对象 
public class Test {
    public static void main(String[] args) {
        //File f1=new File("C:\\Users\\adminys\\Desktop\\ab.txt");
        //File f1=new File("C:/Users/adminys/Desktop/ab.txt");
        File f1=new File("C:"+File.separator+"Users/adminys/Desktop/ab.txt");
        System.out.println(f1.length());//文件大小

        File f2=new File("C:/Users/adminys/Desktop");
        System.out.println(f2.length());//文件夹本身的大小
        //File对象可以指代一个不存在的文件路径
        File f3 = new File("C:/Users/adminys/Desktop/aaa.txt");
        System.out.println(f3.length());//0
        System.out.println(f3.exists());//false
        //定位的文件在模块中
        //绝对路径带盘符
        //File f4=new File("D:\\codetest\\javasepro\\file-io-app\\src\\ys.txt");
        //相对路径，默认是直接工程下寻找文件
        File f4=new File("file-io-app\\src\\ys.txt");
        System.out.println(f4.length());
    };
}

常用方法

public class Test {
    public static void main(String[] args) {
        File f1=new File("C:\\Users\\adminys\\Desktop\\ab.txt");
        //判断当前文件对象，对应的文件路径是否存在
        System.out.println(f1.exists());
        //判断当前文件对象是否是文件
        System.out.println(f1.isFile());
        //判断当前文件对象是否是文件夹
        System.out.println(f1.isDirectory());
        //获取文件名字(包含后缀）
        System.out.println(f1.getName());
        //获取文件大小，返回字节个数
        System.out.println(f1.length());
        //获取文件的最后修改时间
        long time=f1.lastModified();
        SimpleDateFormat sdf=new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
        System.out.println(sdf.format(time));
        //获取创建文件对象时使用的路径
        File f2=new File("C:\\Users\\adminys\\Desktop\\ab.txt");
        File f3=new File("file-io-app\\src\\ys.txt");
        System.out.println(f2.getPath());//C:\Users\adminys\Desktop\ab.txt
        System.out.println(f3.getPath());//file-io-app\src\ys.txt
        //获取绝对路径
        System.out.println(f2.getAbsolutePath());//C:\Users\adminys\Desktop\ab.txt
        System.out.println(f3.getAbsolutePath());//D:\codetest\javasepro\file-io-app\src\ys.txt
    };
}

public class Test {
    public static void main(String[] args) throws IOException {
        //获取当前目录下所有的一级文件名称到一个字符数组中去返回
        File f1=new File("D:/多视图");
        String[] names=f1.list();
        for (String name : names) {
            System.out.println(name);
        }
        //获取当前目录下所有一级文件对象到一个文件对象数组中去返回
        File[] files=f1.listFiles();
        for (File file : files) {
            System.out.println(file.getAbsolutePath());
        }
    };
}

使用listFiles方法注意事项：

  • 当主调是文件，或者当路径不存在，返回null。
  • 当主调是空文件夹时，返回一个长度为0的数组。
  • 当主调是一个有内容的文件夹时，将所有一级文件和文件夹的路径放在一个File数组中返回。
  • 当主调是一个文件夹，且里面有隐藏文件，将所有一级文件和文件夹的路径放在一个File数组中返回，包含隐藏文件。
  • 当主调是一个文件夹，但没有权限访问该文件夹，返回null。

方法递归

  • 直接递归：方法自己调用自己
  • 简介递归：方法调用其他方法，其他方法又调用自己。
public class Test {
    public static void main(String[] args) {
        test1();
    }
    public static void test1(){
        System.out.println("-----test1-----");
        test1();//无限在方法栈中调用test1()，最后栈内存溢出
    }
    public static void test2(){
        System.out.println("-----test2-----");
        test3();
    }
    public static void test3(){
        System.out.println("-----test3-----");
        test2();
    }
}

案例：计算n的阶乘

public class Test {
    public static void main(String[] args) {
        System.out.println(f(5));
    }
    public static int f(int n){
        if(n==1){
            return 1;
        }else{
            return f(n-1)*n;
        }
    }
}

练习案例

文件搜索：

public class Test {
    public static void main(String[] args) throws IOException {
        searchFile(new File("D:/"),"QQMusic.exe");
    }

    /**
     * 去目录下搜索某个文件
     * @param dir
     * @param fileName
     */
    public static void searchFile(File dir, String fileName) throws IOException {
        //把非法情况拦截
        if(dir==null||!dir.exists()||dir.isFile()){
            return;
        }
        //一定是文件夹对象
        File[] files = dir.listFiles();
        if(files==null||files.length==0){
            return;
        }
        for (File file : files) {
            if (file.isFile()) {
                if (file.getName().contains(fileName)) {
                    System.out.println(file.getAbsolutePath());
                    Runtime runtime = Runtime.getRuntime();
                    runtime.exec(file.getAbsolutePath());
                }
            }else{
                searchFile(file,fileName);
            }
        }
    }
}

文件删除：

public class Test {
    public static void main(String[] args) {
        deleteDir(new File("C:\\Users\\adminys\\Desktop\\多视图"));
    }
    public static void deleteDir(File dir) {
        //把非法情况拦截
        if(dir==null||!dir.exists()||dir.isFile()){
            return;
        }
        File[] files = dir.listFiles();
        if(files==null||files.length==0){
            return;
        }
        for(File file:files){
            if(!file.delete()){
                deleteDir(file);
            }
        }
        dir.delete();
    }
}

啤酒2院一瓶，4个盖子可以换一瓶，2个空瓶可以换一瓶，请问10元能换多少瓶？

public class Test {
    public static void main(String[] args) {
        System.out.println(buy(10, 0, 0));
    }
    public static int buy(int money,int bottle,int lid){
        int buynumber=money/2+bottle/2+lid/4;
        bottle=buynumber+bottle%2;
        lid=buynumber+lid%4;
        money=money%2;
        if(money/2+bottle/2+lid/4>0){
            buynumber+=buy(money,bottle,lid);
        }
        return buynumber;
    }
}

字符集

  • ASCII，美国信息交换标准代码，字符集使用一个字节存储字符，首尾是0，总共可以表达128个字符
  • GBK是汉字编码字符集，包含2万个汉字等字符，GBK中一个字符编码成两个字节的形式存储，汉字的第一个字节的第一位必须是0。
  • Unicode字符集是统一码，也叫万国码，是国际组织指定的，可以容纳世界上所有文字符号的字符集。
  • UTF-8是Unicode字符集的一种标准编码，采取可变长编码方案，共分四个长度区：1个字节、2个字节、3个字节、4个字节。英文字符、数字只占1个字节，汉字只占3个字节。
UTF-8编码（二进制）

110xxxxx 10xxxxxx
1110xxxx 10xxxxxx 10xxxxxx
1110xxxx 10xxxxxx 10xxxxxx 10xxxxxx
  • 英文，数字一般不会乱码，因为很多字符集都兼容ASCII编码。
public class Test {
    public static void main(String[] args) throws UnsupportedEncodingException {
       //编码
        String data="a我b";
        byte[] bytes=data.getBytes();//默认按照平台字符集UTF-8进行编码
        System.out.println(Arrays.toString(bytes));//[97, -26, -120, -111, 98]
        //按照指定字符集进行编码
        byte[] bytes1=data.getBytes("GBK");//抛异常
        System.out.println(Arrays.toString(bytes1));//[97, -50, -46, 98]
        //解码
        String s1=new String(bytes);//默认按照平台字符集UTF-8进行解码
        System.out.println(s1);//a我b

        String s2=new String(bytes1);//默认按照平台字符集UTF-8进行解码
        System.out.println(s2);//a��b

        String s3 =new String(bytes1,"GBK");
        System.out.println(s3);//a我b
    }
}
